704  template<typename dtype>
705  dtype det(const NdArray<dtype>& inArray)
706  {
707  const Shape inShape = inArray.shape();
708  if (inShape.rows != inShape.cols)
709  {
710  std::string errStr = "ERROR: linalg::determinant: input array must be square with size no larger than 3x3.";
711  std::cerr << errStr << std::endl;
712  throw std::invalid_argument(errStr);
713  }
714 
715  if (inShape.rows == 1)
716  {
717  return inArray.front();
718  }
719  else if (inShape.rows == 2)
720  {
721  return inArray(0, 0) * inArray(1, 1) - inArray(0, 1) * inArray(1, 0);
722  }
723  else if (inShape.rows == 3)
724  {
725  dtype aei = inArray(0, 0) * inArray(1, 1) * inArray(2, 2);
726  dtype bfg = inArray(0, 1) * inArray(1, 2) * inArray(2, 0);
727  dtype cdh = inArray(0, 2) * inArray(1, 0) * inArray(2, 1);
728  dtype ceg = inArray(0, 2) * inArray(1, 1) * inArray(2, 0);
729  dtype bdi = inArray(0, 1) * inArray(1, 0) * inArray(2, 2);
730  dtype afh = inArray(0, 0) * inArray(1, 2) * inArray(2, 1);
731 
732  return aei + bfg + cdh - ceg - bdi - afh;
733  }
734  else
735  {
736  dtype determinant = 0;
737  NdArray<dtype> submat(inShape.rows - 1);
738 
739  for (uint32 c = 0; c < inShape.rows; ++c)
740  {
741  uint32 subi = 0;
742  for (uint32 i = 1; i < inShape.rows; ++i)
743  {
744  uint32 subj = 0;
745  for (uint32 j = 0; j < inShape.rows; ++j)
746  {
747  if (j == c)
748  {
749  continue;
750  }
751 
752  submat(subi, subj++) = inArray(i, j);
753  }
754  ++subi;
755  }
756  determinant += (static_cast<dtype>(std::pow(-1, c)) * inArray(0, c) * det(submat));
757  }
758 
759  return determinant;
760  }
