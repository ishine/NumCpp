162  {
163  NdArray<double> q0 = inQuat1.toNdArray();
164  NdArray<double> q1 = inQuat2.toNdArray();
165 
166  NdArray<double> qDot = q1 - q0;
167  qDot /= inTime;
168 
169  NdArray<double> eyeTimesScalar(3);
170  eyeTimesScalar.zeros();
171  eyeTimesScalar(0, 0) = inQuat2.s();
172  eyeTimesScalar(1, 1) = inQuat2.s();
173  eyeTimesScalar(2, 2) = inQuat2.s();
174 
175  NdArray<double> epsilonHat = linalg::hat<double>(inQuat2.i(), inQuat2.j(), inQuat2.k());
176  NdArray<double> q(4, 3);
177  q.put(Slice(0, 3), Slice(0, 3), eyeTimesScalar + epsilonHat);
178  q(3, 0) = -inQuat2.i();
179  q(3, 1) = -inQuat2.j();
180  q(3, 2) = -inQuat2.k();

182  NdArray<double> omega = q.transpose().dot(qDot.transpose());

284  static Quaternion fromDCM(const NdArray<dtype>& inDcm)
285  {
286  const Shape inShape = inDcm.shape();
287  if (!(inShape.rows == 3 && inShape.cols == 3))
288  {
289  std::string errStr = "ERROR: rotations::Quaternion::fromDcm: input direction cosine matrix must have shape = (3,3).";
290  std::cerr << errStr << std::endl;
291  throw std::invalid_argument(errStr);
292  }
293 
294  NdArray<double> dcm = inDcm.template astype<double>();
295 
296  NdArray<double> checks(1, 4);
297  checks[0] = dcm(0, 0) + dcm(1, 1) + dcm(2, 2);
298  checks[1] = dcm(0, 0) - dcm(1, 1) + dcm(2, 2);
299  checks[2] = dcm(0, 0) - dcm(1, 1) - dcm(2, 2);
300  checks[3] = dcm(0, 0) + dcm(1, 1) - dcm(2, 2);
301 
302  uint32 maxIdx = argmax(checks).item();
303 
304  double q0 = 0;
305  double q1 = 0;
306  double q2 = 0;
307  double q3 = 0;
308 
309  switch (maxIdx)
310  {
311  case 0:
312  {
313  q3 = 0.5 * std::sqrt(1 + dcm(0, 0) + dcm(1, 1) + dcm(2, 2));
314  q0 = (dcm(1, 2) - dcm(2, 1)) / (4 * q3);
315  q1 = (dcm(2, 0) - dcm(0, 2)) / (4 * q3);
316  q2 = (dcm(0, 1) - dcm(1, 0)) / (4 * q3);
317 
318  break;
319  }
320  case 1:
321  {
322  q2 = 0.5 * std::sqrt(1 - dcm(0, 0) - dcm(1, 1) + dcm(2, 2));
323  q0 = (dcm(0, 2) + dcm(2, 0)) / (4 * q2);
324  q1 = (dcm(1, 2) + dcm(2, 1)) / (4 * q2);
325  q3 = (dcm(0, 1) - dcm(1, 0)) / (4 * q2);
326 
327  break;
328  }
329  case 2:
330  {
331  q0 = 0.5 * std::sqrt(1 + dcm(0, 0) - dcm(1, 1) - dcm(2, 2));
332  q1 = (dcm(0, 1) + dcm(1, 0)) / (4 * q0);
333  q2 = (dcm(0, 2) + dcm(2, 0)) / (4 * q0);
334  q3 = (dcm(1, 2) - dcm(2, 1)) / (4 * q0);
335 
336  break;
337  }
338  case 3:
339  {
340  q1 = 0.5 * std::sqrt(1 - dcm(0, 0) + dcm(1, 1) - dcm(2, 2));
341  q0 = (dcm(0, 1) + dcm(1, 0)) / (4 * q1);
342  q2 = (dcm(1, 2) + dcm(2, 1)) / (4 * q1);
343  q3 = (dcm(2, 0) - dcm(0, 2)) / (4 * q1);
344 
345  break;
346  }
347  }

349  return Quaternion(q0, q1, q2, q3);

456  static Quaternion slerp(const Quaternion& inQuat1, const Quaternion& inQuat2, double inPercent)
457  {
458  if (inPercent < 0 || inPercent > 1)
459  {
460  std::string errStr = "ERROR: rotations::Quaternion::slerp: input percent must be of the range [0, 1]";
461  std::cerr << errStr << std::endl;
462  throw std::invalid_argument(errStr);
463  }
464 
465  if (inPercent == 0)
466  {
467  return inQuat1;
468  }
469  else if (inPercent == 1)
470  {
471  return inQuat2;
472  }
473 
474  double dotProduct = dot<double>(inQuat1.toNdArray(), inQuat2.toNdArray()).item();
475 
476  // If the dot product is negative, the quaternions
477  // have opposite handed-ness and slerp won't take
478  // the shorter path. Fix by reversing one quaternion.
479  Quaternion quat1Copy(inQuat1);
480  if (dotProduct < 0.0)
481  {
482  quat1Copy *= -1;
483  dotProduct *= -1;
484  }
485 
486  const double DOT_THRESHOLD = 0.9995;
487  if (dotProduct > DOT_THRESHOLD) {
488  // If the inputs are too close for comfort, linearly interpolate
489  // and normalize the result.
490  return nlerp(inQuat1, inQuat2, inPercent);
491  }
492 
493  dotProduct = clip(dotProduct, -1.0, 1.0); // Robustness: Stay within domain of acos()
494  const double theta0 = std::acos(dotProduct); // angle between input vectors
495  const double theta = theta0 * inPercent; // angle between v0 and result
496 
497  const double s0 = std::cos(theta) - dotProduct * std::sin(theta) / std::sin(theta0); // == sin(theta_0 - theta) / sin(theta_0)
498  const double s1 = std::sin(theta) / std::sin(theta0);
499 
500  NdArray<double> interpQuat = (inQuat1.toNdArray() * s0) + (inQuat2.toNdArray() * s1);
501  return Quaternion(interpQuat);
502  }

541  {
542  NdArray<double> dcm(3);
543 
544  double q0 = i();
545  double q1 = j();
546  double q2 = k();
547  double q3 = s();
548 
549  dcm(0, 0) = utils::sqr(q3) + utils::sqr(q0) - utils::sqr(q1) - utils::sqr(q2);
550  dcm(0, 1) = 2 * (q0 * q1 + q3 * q2);
551  dcm(0, 2) = 2 * (q0 * q2 - q3 * q1);
552  dcm(1, 0) = 2 * (q0 * q1 - q3 * q2);

554  dcm(1, 2) = 2 * (q1 * q2 + q3 * q0);
555  dcm(2, 0) = 2 * (q0 * q2 + q3 * q1);
556  dcm(2, 1) = 2 * (q1 * q2 - q3 * q0);
557  dcm(2, 2) = utils::sqr(q3) - utils::sqr(q0) - utils::sqr(q1) + utils::sqr(q2);;
558 
559  return dcm;
