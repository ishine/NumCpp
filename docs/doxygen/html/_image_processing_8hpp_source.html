729  // Description:
732  void runClusterMaker()
733  {
734  uint32 clusterId = 0;
735 
736  for (auto& currentPixel : xcdsVec_)
737  {
738  // not already visited
739  if (currentPixel.clusterId() == -1)
740  {
741  Cluster<dtype> newCluster(clusterId); // a new cluster
742  currentPixel.setClusterId(clusterId);
743  newCluster.addPixel(currentPixel); // assign pixel to cluster
744 
745  // get the neighbors
746  std::vector<uint32> neighborIds;
747  findNeighborXcds(currentPixel, neighborIds);
748  if (neighborIds.empty())
749  {
750  clusters_.push_back(newCluster);
751  ++clusterId;
752  continue;
753  }
754 
755  // loop through the neighbors
756  for (uint32 neighborsIdx = 0; neighborsIdx < neighborIds.size(); ++neighborsIdx)
757  {
758  Pixel<dtype>& currentNeighborPixel = xcdsVec_[neighborIds[neighborsIdx]];
759 
760  // go to neighbors
761  std::vector<uint32> newNeighborIds;
762  findNeighborXcds(currentNeighborPixel, newNeighborIds);
763 
764  // loop through the new neighbors and add them to neighbors
765  for (auto newNeighborId : newNeighborIds)
766  {
767  // not already in neighbors
768  if (std::find(neighborIds.begin(), neighborIds.end(), newNeighborId) == neighborIds.end())
769  {
770  neighborIds.push_back(newNeighborId);
771  }
772  }
773 
774  // not already assigned to a cluster
775  if (currentNeighborPixel.clusterId() == -1)
776  {
777  currentNeighborPixel.setClusterId(clusterId);
778  newCluster.addPixel(currentNeighborPixel);
779  }
780  }
781 
782  clusters_.push_back(newCluster);
783  ++clusterId;
784  }
785  }

950  // Description:
958  void centerOfMass(const Cluster<dtype>& inCluster)
959  {
960  const Shape clusterShape(inCluster.height(), inCluster.width());
961  NdArray<dtype> clusterArray(clusterShape);
962  clusterArray.zeros();
963 
964  const uint32 rowMin = inCluster.rowMin();
965  const uint32 colMin = inCluster.colMin();
966  const dtype inten = inCluster.intensity();
967 
968  for (auto& pixel : inCluster)
969  {
970  clusterArray(pixel.row() - rowMin, pixel.col() - colMin) = pixel.intensity();
971  }
972 
973  // first get the row center
974  row_ = 0;
975  uint32 theRow = rowMin;
976  for (uint32 rowIdx = 0; rowIdx < clusterShape.rows; ++rowIdx)
977  {
978  double rowSum = 0;
979  for (uint32 colIdx = 0; colIdx < clusterShape.cols; ++colIdx)
980  {
981  rowSum += static_cast<double>(clusterArray(rowIdx, colIdx));
982  }
983  row_ += rowSum * static_cast<double>(theRow++);
984  }
985 
986  row_ /= static_cast<double>(inten);
987 
988  // then get the column center
989  col_ = 0;
990  uint32 theCol = colMin;
991  for (uint32 colIdx = 0; colIdx < clusterShape.cols; ++colIdx)
992  {
993  double colSum = 0;
994  for (uint32 rowIdx = 0; rowIdx < clusterShape.rows; ++rowIdx)
995  {
996  colSum += static_cast<double>(clusterArray(rowIdx, colIdx));
997  }
998  col_ += colSum * static_cast<double>(theCol++);
999  }
1000 
1001  col_ /= static_cast<double>(inten);
1002  }

1296  {

1298  {
1299  std::string errStr = "ERROR: imageProcessing::generateThreshold: input rate must be of the range [0, 1]";
1300  std::cerr << errStr << std::endl;
1301  throw std::invalid_argument(errStr);
1302  }
1303 
1304  // first build a histogram
1305  int32 minValue = static_cast<int32>(std::floor(inImageArray.min().item()));
1306  int32 maxValue = static_cast<int32>(std::floor(inImageArray.max().item()));
1307 
1308  if (utils::essentiallyEqual(inRate, 0.0))
1309  {
1310  return static_cast<dtype>(maxValue);
1311  }
1312  else if (utils::essentiallyEqual(inRate, 1.0))
1313  {
1314  if (DtypeInfo<dtype>::isSigned())
1315  {
1316  return static_cast<dtype>(minValue - 1);
1317  }
1318  else
1319  {
1320  return static_cast<dtype>(0);
1321  }
1322  }
1323 
1324  const uint32 histSize = static_cast<uint32>(maxValue - minValue + 1);
1325 
1326  NdArray<double> histogram(1, histSize);
1327  histogram.zeros();
1328  for (auto intensity : inImageArray)
1329  {
1330  const uint32 bin = static_cast<uint32>(static_cast<int32>(std::floor(intensity)) - minValue);
1331  ++histogram[bin];
1332  }
1333 
1334  // integrate the normalized histogram from right to left to make a survival function (1 - CDF)
1335  const double dNumPixels = static_cast<double>(inImageArray.size());
1336  NdArray<double> survivalFunction(1, histSize + 1);
1337  survivalFunction[-1] = 0;
1338  for (int32 i = histSize - 1; i > -1; --i)
1339  {
1340  double histValue = histogram[i] / dNumPixels;
1341  survivalFunction[i] = survivalFunction[i + 1] + histValue;
1342  }
1343 
1344  // binary search through the survival function to find the rate
1345  uint32 indexLow = 0;
1346  uint32 indexHigh = histSize - 1;
1347  uint32 index = indexHigh / 2; // integer division
1348 
1349  const bool keepGoing = true;
1350  while (keepGoing)
1351  {
1352  const double value = survivalFunction[index];
1353  if (value < inRate)
1354  {
1355  indexHigh = index;
1356  }
1357  else if (value > inRate)
1358  {
1359  indexLow = index;
1360  }
1361  else
1362  {
1363  const int32 thresh = static_cast<int32>(index) + minValue - 1;
1364  if (DtypeInfo<dtype>::isSigned())
1365  {
1366  return static_cast<dtype>(thresh);
1367  }
1368  else
1369  {
1370  return thresh < 0 ? 0 : static_cast<dtype>(thresh);
1371  }
1372  }
1373 
1374  if (indexHigh - indexLow < 2)
1375  {
1376  return static_cast<dtype>(static_cast<int32>(indexHigh) + minValue - 1);
1377  }
1378 
1379  index = indexLow + (indexHigh - indexLow) / 2;
1380  }
1381 
1382  // shouldn't ever get here but stop the compiler from throwing a warning
1383  return static_cast<dtype>(histSize - 1);
1384  }
